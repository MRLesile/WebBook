# 动态设置XRSetting中不同设备所需的 XRLoader

   [MenuItem("Tools/测试XR")]
    /// <summary>
    /// 动态设置XRPlugins
    /// </summary>
    public static void SetXRPlugins()
    {
        XRGeneralSettings androidXRSettings =
            XRGeneralSettingsPerBuildTarget.XRGeneralSettingsForBuildTarget(BuildTargetGroup.Android);

        if (androidXRSettings == null)
        {
            var assignedSettings = ScriptableObject.CreateInstance<XRManagerSettings>() as XRManagerSettings;
            androidXRSettings.AssignedSettings = assignedSettings;
            EditorUtility.SetDirty(androidXRSettings); 
        }

        //Pico loaderTypeName：Unity.XR.PXR.PXR_Loader
        //Oculus LoaderTypeName:Unity.XR.Oculus.OculusLoader

        IReadOnlyList<XRLoader> list = androidXRSettings.Manager.activeLoaders;
        for (int i = 0; i < list.Count; i++)
        {
            string nameTemp = list[i].GetType().FullName;
            //取消
            XRPackageMetadataStore.RemoveLoader(androidXRSettings.Manager, nameTemp, BuildTargetGroup.Android);
        }
        string loaderTypeName;
        if (isPico)
        {
            //Pico
            loaderTypeName = "Unity.XR.PXR.PXR_Loader";
        }
        else
        {
            //Oculus
            loaderTypeName = "Unity.XR.Oculus.OculusLoader";
        }
        XRPackageMetadataStore.AssignLoader(androidXRSettings.Manager, loaderTypeName, BuildTargetGroup.Android);
    }
  • 可以配合OnPreprocessBuild 来打包前自动设置防止切换Pico 跟Oculus是忘记切换。宏定义也一样以下是完整代码
using UnityEditor;
using UnityEditor.Build;
using UnityEditor.Build.Reporting;
using System;
using System.Collections.Generic;
using UnityEditor.XR.Management;
using UnityEditor.XR.Management.Metadata;
using UnityEngine;
using UnityEngine.XR;
using UnityEngine.XR.Management;

public class PreBuildCustomization : IPreprocessBuildWithReport
{
    private static bool isPico = false;

    public int callbackOrder
    {
        get { return 0; }
    }

    public void OnPreprocessBuild(BuildReport report)
    {
        if (report.summary.platform == BuildTarget.Android)
        {
            // 显示弹出窗口询问用户选择平台
            isPico = EditorUtility.DisplayDialog("当前设备打包选择", "请问是Pico设备还是Oculus设备:", "PICO", "OCULUS");

            SetXRPlugins();
            
            if (isPico)
            {
                Debug.Log("Selected platform: Pico");
                // 在这里执行Pico平台的设置修改
                ModifyDefineSymbol("OCULUS", false);
                ModifyDefineSymbol("PICO", true);
            }
            else
            {
                Debug.Log("Selected platform: Oculus");
                // 在这里执行Oculus平台的设置修改
                ModifyDefineSymbol("OCULUS", true);
                ModifyDefineSymbol("PICO", false);
            }
        }
    }

    static void ModifyDefineSymbol(string symbol, bool enable)
    {
        List<string> defines =
            new List<string>(PlayerSettings.GetScriptingDefineSymbolsForGroup(BuildTargetGroup.Android).Split(';'));

        if (enable)
        {
            if (!defines.Contains(symbol))
            {
                defines.Add(symbol);
            }
        }
        else
        {
            if (defines.Contains(symbol))
            {
                defines.Remove(symbol);
            }
        }

        PlayerSettings.SetScriptingDefineSymbolsForGroup(BuildTargetGroup.Android, string.Join(";", defines.ToArray()));
    }

    /// <summary>
    /// 设置VRSetting的平台如Oculus跟Pico
    /// </summary>
    static void CheckVRSettingPlatom()
    {
        var XrSetting = XRGeneralSettings.Instance;

        if (XrSetting == null)
        {
            Debug.LogError("XR General Settings is not available");
        }
    }


    [MenuItem("Tools/测试XR")]
    /// <summary>
    /// 动态设置XRPlugins
    /// </summary>
    public static void SetXRPlugins()
    {
        XRGeneralSettings androidXRSettings =
            XRGeneralSettingsPerBuildTarget.XRGeneralSettingsForBuildTarget(BuildTargetGroup.Android);

        if (androidXRSettings == null)
        {
            var assignedSettings = ScriptableObject.CreateInstance<XRManagerSettings>() as XRManagerSettings;
            androidXRSettings.AssignedSettings = assignedSettings;
            EditorUtility.SetDirty(androidXRSettings); 
        }

        //Pico loaderTypeName：Unity.XR.PXR.PXR_Loader
        //Oculus LoaderTypeName:Unity.XR.Oculus.OculusLoader

        IReadOnlyList<XRLoader> list = androidXRSettings.Manager.activeLoaders;
        for (int i = 0; i < list.Count; i++)
        {
            string nameTemp = list[i].GetType().FullName;
            //取消
            XRPackageMetadataStore.RemoveLoader(androidXRSettings.Manager, nameTemp, BuildTargetGroup.Android);
        }
        string loaderTypeName;
        if (isPico)
        {
            //Pico
            loaderTypeName = "Unity.XR.PXR.PXR_Loader";
        }
        else
        {
            //Oculus
            loaderTypeName = "Unity.XR.Oculus.OculusLoader";
        }
        XRPackageMetadataStore.AssignLoader(androidXRSettings.Manager, loaderTypeName, BuildTargetGroup.Android);
    }
}
